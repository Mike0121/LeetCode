```python
class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        result = []
        task_stack = [(0, target, [])]

        while task_stack:
            index, current_target, combination = task_stack.pop()

            if index >= len(candidates):
                if current_target == 0:
                    result.append(combination)
                continue

            if candidates[index] <= current_target:
                task_stack.append((index, current_target - candidates[index], combination + [candidates[index]]))

            task_stack.append((index + 1, current_target, combination))

        return result
```
